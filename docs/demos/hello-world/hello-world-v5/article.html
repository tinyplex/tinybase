Hello World v5

In this demo, we use Svelte to render data in the Store object and then change the Cell to see the display update. We're making changes to the Hello World v4 demo.

We swap the React imports for Svelte and the TinyBase Svelte module:

 <script type="importmap">
   {
     "imports": {
       "tinybase": "https://esm.sh/tinybase@8.1.0-beta.5",
-      "tinybase/ui-react": "https://esm.sh/tinybase@8.1.0-beta.5/ui-react",
-      "tinybase/ui-react-inspector": "https://esm.sh/tinybase@8.1.0-beta.5/ui-react-inspector",
-      "react": "https://esm.sh/react@^19.2.4",
-      "react/jsx-runtime": "https://esm.sh/react@^19.2.4/jsx-runtime",
-      "react-dom/client": "https://esm.sh/react-dom@^19.2.4/client"
+      "tinybase/ui-svelte": "https://esm.sh/tinybase@8.1.0-beta.5/ui-svelte",
+      "svelte": "https://esm.sh/svelte@^5.55.1"
     }
   }
 </script>

The main entry point no longer needs the React imports:

 import {createStore} from 'tinybase';
-import {CellView, Provider} from 'tinybase/ui-react';
-import {Inspector} from 'tinybase/ui-react-inspector';
-import {createRoot} from 'react-dom/client';
-import React from 'react';
+import {mount} from 'svelte';
+import App from './App.svelte';

Instead of rendering a React tree, we mount a Svelte component and pass the Store in as a prop:

-createRoot(document.body).render(
-  <Provider store={store}>
-    <CellView tableId="t1" rowId="r1" cellId="c1" />
-    <Inspector />
-  </Provider>,
-);
+mount(App, {target: document.body, props: {store}});

The Svelte component uses the Provider and CellView components from the ui-svelte module:

<script>
  import {CellView, Provider} from 'tinybase/ui-svelte';

  let {store} = $props();
</script>

<Provider {store}>
  <CellView tableId="t1" rowId="r1" cellId="c1" />
</Provider>

Unlike the React version, there is not yet a Svelte Inspector component, so this demo just focuses on the live-updating Cell itself.
